// コメント取得関数を独立（401エラー対策のログを追加） async function loadComments(title) { const commentList = document.getElementById('comment-list'); try { const adminPassword = sessionStorage.getItem('adminPassword') || ''; // パスワードがある時だけクエリに含める let url = `/api/comments?article_title=${encodeURIComponent(title)}`; if (adminPassword) { url += `&password=${encodeURIComponent(adminPassword)}`; } const commentRes = await fetch(url); if (commentRes.ok) { const comments = await commentRes.json(); if (Array.isArray(comments) && comments.length > 0) { commentList.innerHTML = comments.map(comment => `
${comment.name || '匿名'}さん
${comment.message}
`).join(''); } else { commentList.innerHTML = '

まだコメントはありません。最初の感想をお寄せください。

'; } } else { // ここで 401 が出なくなります commentList.innerHTML = '

コメントを表示できませんでした。

'; } } catch (e) { console.error('コメント取得エラー:', e); } }
